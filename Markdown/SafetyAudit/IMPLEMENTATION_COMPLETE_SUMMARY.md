export const getAllActiveRequirements = async (): Promise<any[]> => {
  const { data, error } = await supabase
    .from('v_active_audit_requirements')
    .select('*')
    .order('category_identifier, display_order, item_number');

  if (error) throw error;
  return data || [];
};
